{{/* ====================================== TOPICS ====================================== */}} {{/* This outputs the topics used on a specific page.*/}} {{/* Checks if there are topics applied to this page */}} {{- if (not (isset .Params "topics")) | or (eq .Params.topics "") -}} {{- else -}}
{{/* Sets the taxonomy we're pulling here */}} {{- $taxonomy := "topics" -}} {{- $topics := .Param $taxonomy -}} {{/* Gets the number of topics that are applied */}} {{- $length := $topics | len -}} {{/* For each of the topics,... */}} {{/* NOTE: Many of the pages in digital.gov have duplicate topics. */}} {{/* So we are running "uniq" on the array to remove duplicates */}} {{- range $i, $e := uniq (sort $topics) -}} {{- $slug := . -}} {{/* Gets the Topics page */}} {{- with $.Site.GetPage (printf "/%s/%s" $taxonomy $slug) -}} {{- if or (eq .Params.weight 2) (eq .Params.weight 3) -}} {{- .Title -}} {{- end -}} {{- end -}} {{- end -}}
{{- end -}}
